1086 inline static int32_t twosComplement(int32_t raw_value, const int bit_length) {
1087 // Check if the sign bit is set by shifting to the sign bit position
1088 if (raw_value & (1 << (bit_length - 1))) {
1089 // Perform sign extension by setting upper bits to 1
1090 raw_value |= ~((1 << bit_length) - 1);
1091 } else {
1092 // Clear upper bits to zero for positive numbers
1093 raw_value &= (1 << bit_length) - 1;
1094 }
1095 return raw_value;
