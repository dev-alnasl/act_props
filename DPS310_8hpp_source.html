1143 static inline int32_t twosComplement(int32_t raw_value, const int bit_length) {
1144 // Check if the sign bit is set by shifting to the sign bit position
1145 if (raw_value & (1U << (bit_length - 1))) {
1146 // Perform sign extension by setting upper bits to 1
1147 raw_value |= ~((1U << bit_length) - 1);
1148 } else {
1149 // Clear upper bits to zero for positive numbers
1150 raw_value &= (1U << bit_length) - 1;
1151 }
1152 return raw_value;
