663 static inline int32_t twosComplement(int32_t raw_value, const int bit_length) {
664 // Check if the sign bit is set by shifting to the sign bit position
665 if (raw_value & (1U << (bit_length - 1))) {
666 // Perform sign extension by setting upper bits to 1
667 raw_value |= ~((1U << bit_length) - 1);
668 } else {
669 // Clear upper bits to zero for positive numbers
670 raw_value &= (1U << bit_length) - 1;
671 }
